Welcome to the Outlet REPL

Outlet is a new language to experiment with compiling Lisp to other languages like javascript and lua. Read more about it here and download it here.

Type some outlet below:
Target:

result

output

Examples

(define foo 5)

(define (bar x)
  (+ x 1))

(define (buz x)
  (let ((y x)
        (add_1 (if (< x 0)
                   (lambda (n) (- n 1))
                   (lambda (n) (+ n 1)))))
    (add_1 y)))

(define (fib n)
  (if (= n 0)
      n
      (+ n (fib (- n 1)))))

(define (Y func)
  ((lambda (f) (f f))
   (lambda (f)
     (func f))))

(define (string_split str c)
  (str.split c))

(string_split "hello\"world" "\"")

(for-each
 (lambda (x) (console.log x))
 '(1 2 3 4))
